on reset goes to FI
FI:
  set finished = 0
  next FI if ifu_respValid == 0
  next EI otherwise
EI:
  set lsu_reqValid = 1, next LW if inst_type == Load
  set lsu_reqValid = 1, next SW if inst_type == Store
  next RW otherwise
LW:
  next LW if lsu_respValid == 0
  next RW otherwise
SW:
  next SW if lsu_respValid == 0
  set finished = 1, next FI otherwise
RW:
  set finished = 1, next FI
